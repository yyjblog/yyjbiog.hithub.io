第一篇 c++重载

    #include 
        using namespace std;
        
        class Distance
        {
        private:
            int feet;             // 0 到无穷
            int inches;           // 0 到 12
        public:
            // 所需的构造函数
            Distance() {
                feet = 0;
                inches = 0;
            }
            Distance(int f, int i) {
                feet = f;
                inches = i;
            }
            // 显示距离的方法
            void displayDistance()
            {
                cout << "F: " << feet << " I:" << inches << endl;
            }
            // 重载负运算符--前缀
            Distance operator-- ()
            {
                --feet;
                --inches;
                return Distance(feet, inches);
            }
            //重载负运算符--后缀
            Distance operator--(int) {
        
                Distance temp(feet, inches);
                --feet;
                --inches;
          //也可以直接这样换  --*this;
                return temp;
            }
        };
        int main()
        {
            Distance D1(11, 10), D2(-5, 11);
        
            (--D1).displayDistance();                     
            D1.displayDistance();    
        
            D2--.displayDistance();                     
            D2.displayDistance();    
        
            return 0;
        }
        代码是这样的，应该c++里一元运算符默认都是起在变量前面的，并且大多没有参数，因为单个数据参与运算大多就是类里的那个参与运算。
        而像++，--这样的起在标识符后面的往往会在后莫按摩加一个变量？可能是为了区分，这个变量本身并不参与函数内的任何运算，而运算先后顺序就由函数自己来定了。
