        using namespace std;
        
        class Distance
        {
        private:
            int feet;             // 0 到无穷
            int inches;           // 0 到 12
        public:
            // 所需的构造函数
            Distance() {
                feet = 0;
                inches = 0;
            }
            Distance(int f, int i) {
                feet = f;
                inches = i;
            }
            // 显示距离的方法
            void displayDistance()
            {
                cout << "F: " << feet << " I:" << inches << endl;
            }
            // 重载负运算符--前缀
            Distance operator-- ()
            {
                --feet;
                --inches;
                return Distance(feet, inches);
            }
            //重载负运算符--后缀
            Distance operator--(int) {
        
                Distance temp(feet, inches);
                --feet;
                --inches;
          //也可以直接这样换  --*this;
                return temp;
            }
        };
        int main()
        {
            Distance D1(11, 10), D2(-5, 11);
        
            (--D1).displayDistance();                     
            D1.displayDistance();    
        
            D2--.displayDistance();                     
            D2.displayDistance();    
        
            return 0;
        }
